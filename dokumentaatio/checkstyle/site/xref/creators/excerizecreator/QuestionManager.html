5   public class QuestionManager {
6       public List<Question> questions;
7       
8       public QuestionManager() {
9           this.questions = new ArrayList<>();
10      }
11      
12      public void createMC(String question, String answer, List<String> choices) {
13          MultipleChoice mc = new MultipleChoice(question, answer, choices);
14          this.questions.add(mc);
15      }
16      
17      public void createOQ(String question, String answer) {
18          OpenQuestion oq = new OpenQuestion(question, answer);
19          this.questions.add(oq);
20      }
21      
22      public void createEssay(String question, String answer) {
23          Essay es = new Essay(question, answer);
24          this.questions.add(es);
25      }
26      
27      public void answer(String question, String answered) {
28          for (Question q : this.questions) {
29              if (q.returnQuestion().equals(question)) {
30                  q.addAnswered(answered);
31              }
32          }
33      }
34      
35      public List<Question> returnQuestions() {
36          return this.questions;
37      }
38      
39      public void checkIfCorrect() {
40          for (Question q : this.questions) {
41              if (q.returnAnswer().equals(q.returnAnswered())) {
42                  q.correct();
43              } else {
44                  q.incorrect();
45              }
46          }
47      }
