1   package creators.excersizecreator.questionlogic;
2   
3   import creators.excersizecreator.questions.OpenQuestion;
4   import creators.excersizecreator.questions.MultipleChoice;
5   import creators.excersizecreator.questions.Question;
6   import creators.excersizecreator.questions.Essay;
7   import java.util.*;
8   
9   public class QuestionManager {
10      public List<Question> questions;
11      
12      public QuestionManager() {
13          this.questions = new ArrayList<>();
14      }
15      
16      public void createMC(String question, String notes, Map<String, Boolean> choices) {
17          MultipleChoice mc = new MultipleChoice(question, notes, choices);
18          this.questions.add(mc);
19      }
20      
21      public void createOQ(String question, String notes, List<String> answer) {
22          OpenQuestion oq = new OpenQuestion(question, notes, answer);
23          this.questions.add(oq);
24      }
25      
26      public void createEssay(String question, String notes, List<String> answer) {
27          Essay es = new Essay(question, notes, answer);
28          this.questions.add(es);
29      }
30      
31      public void answer(String question, String answered) {
32          for (Question q : this.questions) {
33              if (q.returnQuestion().equals(question)) {
34                  q.addAnswered(question, answered);
35              }
36          }
37      }
38      
39      public List<Question> returnQuestions() {
40          return this.questions;
41      }
42      
43      public List<String> returnQuestionStrings() {
44          List<String> qs = new ArrayList<>();
45          for (Question q : this.questions) {
46              qs.add(q.returnQuestion());
47          }
48          return qs;
49      }
50      
51      public void checkIfCorrect() {
52          for (Question q : this.questions) {
53              q.check();
54          }
55      }
56  }
